import random def print_board(board): for row in board: print(" ".join(row)) print() def check_winner(board): for row in board: if row[0] == row[1] == row[2] and row[0] != " ": return row[0] for col in range(3): if board[0][col] == board[1][col] == board[2][col] and board[0][col] != " ": return board[0][col] if board[0][0] == board[1][1] == board[2][2] and board[0][0] != " ": return board[0][0] if board[0][2] == board[1][1] == board[2][0] and board[0][2] != " ": return board[0][2] return None def is_full(board): return all(cell != " " for row in board for cell in row) def random_ai_move(board): empty_cells = [(i, j) for i in range(3) for j in range(3) if board[i][j] == " "] return random.choice(empty_cells) if empty_cells else (-1, -1) def smart_ai_move(board): if random.random() < 0.5: # 50% chance to play a random move for unpredictability return random_ai_move(board) best_score = float('-inf') move = (-1, -1) for i in range(3): for j in range(3): if board[i][j] == " ": board[i][j] = "O" score = random.randint(-10, 10) # Introduce randomness in move scoring board[i][j] = " " if score > best_score: best_score = score move = (i, j) return move def player_move(board): while True: try: row, col = map(int, input("Enter your move (row and column 0-2): ").split()) if board[row][col] == " ": board[row][col] = "X" break else: print("Cell already taken!") except (ValueError, IndexError): print("Invalid input! Enter numbers between 0 and 2.") def play_game(): board = [[" " for _ in range(3)] for _ in range(3)] print("Tic Tac Toe - You are X, AI is O") print_board(board) for turn in range(9): if turn % 2 == 0: player_move(board) else: print("AI is thinking...") row, col = smart_ai_move(board) board[row][col] = "O" print_board(board) winner = check_winner(board) if winner: print(f"{winner} wins!") return print("It's a draw!") if __name__ == "__main__": play_game()
